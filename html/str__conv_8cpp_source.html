74// Converts std::u8string (UTF-8) to std::string (locale multibyte)
75std::string
76u8string_to_string( const std::u8string& u8str )
77{
78#if 1

80#else
81 std::string result;
82 mbstate_t state { }; // Initial conversion state (zero-initialized)
83 const char8_t* current = u8str.data( );
84 const char8_t* end = u8str.data( ) + u8str.size( );
85 char mb_buf[ MB_CUR_MAX ]; // Buffer for multibyte output (max size for
86 // locale)
87
88 // Process each UTF-8 code unit
89 while ( current < end )
90 {
91 // Convert char8_t to multibyte sequence
92 size_t bytes_written = std::c8rtomb( mb_buf, *current, &state );
93
94 if ( bytes_written == static_cast< size_t >( -1 ) )
95 {
96 throw std::runtime_error(
97 "Failed to convert UTF-8 to multibyte (invalid sequence)" );
98 }
99 else if ( bytes_written > 0 )
100 {
101 // Successfully generated multibyte bytes; append to result
102 result.append( mb_buf, bytes_written );
103 ++current;
104 }
105 else
106 { // bytes_written == 0: need more input to complete conversion
107 ++current; // Consume current code unit and continue
108 }
109 }
110
111 // Flush remaining state (e.g., shift sequences for some encodings)
112 size_t bytes_written =
113 std::c8rtomb( mb_buf, 0, &state ); // Null char to flush
114 if ( bytes_written == static_cast< size_t >( -1 ) )
115 {
116 throw std::runtime_error(
117 "Failed to flush state (incomplete UTF-8 sequence)" );
118 }
119 if ( bytes_written > 0 ) { result.append( mb_buf, bytes_written ); }
120
121 return result;

134string_to_u8string( const std::string& str )
135{
136#if 1
137 return ::utf8_to_char8( str );
138#else
139 std::u8string result;
140 mbstate_t state { }; // Initial conversion state
141 const char* current = str.data( );
142 const char* end = str.data( ) + str.size( );
143 char8_t c8; // Output UTF-8 code unit
144
145 while ( current < end )
146 {
147 // Convert multibyte sequence to UTF-8 code unit
148 size_t bytes_consumed =
149 std::mbrtoc8( &c8, current, end - current, &state );
150
151 if ( bytes_consumed == static_cast< size_t >( -1 ) )
152 {
153 throw std::runtime_error(
154 "Failed to convert multibyte to UTF-8 (invalid sequence)" );
155 }
156 else if ( bytes_consumed == static_cast< size_t >( -2 ) )
157 {
158 throw std::runtime_error( "Incomplete multibyte sequence" );
159 }
160 else if ( bytes_consumed == 0 )
161 {
162 // Null character; skip or handle as needed
163 current += 1;
164 }
165 else
166 {
167 // Append the UTF-8 code unit to result
168 result.push_back( c8 );
169 current += bytes_consumed;
170 }
171 }
172
173 // Verify no incomplete sequences remain
174 if ( std::mbrtoc8( nullptr, nullptr, 0, &state ) != 0 )
175 {
176 throw std::runtime_error( "Incomplete multibyte sequence at end" );
177 }

180#endif
