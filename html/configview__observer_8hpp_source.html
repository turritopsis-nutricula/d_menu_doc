171 void disconnect( void* id )
172 {
173 if ( id == nullptr ) return;
174
175 // parent search
176 {
177 decltype( connects_ )::iterator iter;
178 for ( ;; )
179 {
180 iter =
181 std::ranges::find_if( connects_, [ id ]( auto const& i ) {
182 auto const& parent = std::get< parent_id_idx >( i );
183 return parent.has_value( ) && ( *parent ) == id;
184 } );
185
186 if ( iter == connects_.end( ) ) break;
187
188 disconnect( std::get< id_idx >( *iter ) );
189 }
190 }
191
192 // curent disconnect
193 {
194 auto filter_func = [ & ]( auto& i ) {
195 return std::get< id_idx >( i ) == id;
196 };
197
198 auto r = std::ranges::remove_if( connects_, filter_func );
199 for ( auto& i : r ) std::get< connection_idx >( i ).disconnect( );
200
201 connects_.erase( r.begin( ), connects_.end( ) );
202 }
