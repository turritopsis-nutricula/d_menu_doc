209template< typename... Ts >
210auto
211create_new_common_action(
212 domain::commands::common_command_kind kind,
213 repositories::transaction< Ts... >& tran,
214 std::optional< domain::app::config >& cfg )
215{
216
217 using domain::commands::common_command_kind;
218 using domain::commands::parameter_kind;
219
220 auto ser =
221 [ kind, &tran, &cfg ] -> dd::services::create_action_domain_service {
222 switch ( kind )
223 {
224 case common_command_kind::shellexecute :
225 return {
226 *cfg,
227 get< domain::commands::invoker >( tran ),
228 domain::commands::buildin::shellexecute_command_parameter(
229 L"",
230 L"" ) };
231 case common_command_kind::open_sub_menu :
232 return {
233 *cfg,
234 get< domain::commands::invoker >( tran ),
235 domain::commands::buildin::open_sub_menu_command_parameter(
236 domain::menu::menu_id::nil_id( ) ) };
237 default :
238 return {
239 *cfg,
240 get< domain::commands::invoker >( tran ),
241 domain::commands::buildin::nop_command_parameter( ) };
242 }
243 }( );
244
245 ser.update_config( *cfg );
246 update_entity( tran, *cfg );
247
248 auto [ new_act, new_com ] = ser.result( );
249
250 create_entity( tran, new_act );
251 create_entity( tran, new_com );
252 return new_act.entity_id( );
253}
