106v002::backup_json
107to_v002_converter::operator( )( v001::backup_json&& v1 ) const
108{
109
110 v002::backup_json v2;
111
112 v2 = v1;
113
114 // close_menu_command_id recalc
115 auto close_com_list = v2.data.commands |
116 std::views::filter( []( auto const& c ) {
117 return c.command_kind == "close_menu";
118 } ) | std::views::filter( []( auto const& c ) {
119 return c.parameters.size( ) > 0u;
120 } );
121
122 for ( auto& menu : v2.data.menus )
123 {
124 auto const menu_id = menu.id;
125 auto iter = std::ranges::find_if(
126 close_com_list,
127 [ &menu_id ]( auto const& c ) {
128 auto const& p = c.parameters;
129 return std::visit( uuid_compare_visiter( menu_id ), p[ 0 ].value );
130 } );
131
132 if ( iter != close_com_list.end( ) )
133 {
134 menu.close_command_id = iter->id;
135 }
136 }
137
138 return v2;
139}
