473 void insert_base_sync(
474 std::vector< V > const& new_list,
475 DPred&& compare_pred,
476 Converter&& converter,
477 Updater&& updater )
478 {
479
480 auto& l = implp_->list_;
481 if ( new_list.size( ) == 0u )
482 {
483 clear( );
484 return;
485 }
486 // if ( list_.size( ) < new_list.size( ) )
487 // list_.resize( new_list.size( ) );
488
489 // move phase loop
490
491 auto current_iter = l.begin( );
492 for ( auto new_index = 0u; new_index < new_list.size( ); ++new_index )
493 {
494#if 0
495 auto new_value = std::move( new_list[ new_index ] );
496#else
497 auto new_value = new_list[ new_index ];
498#endif
499
500 // foword find
501 auto i = std::find_if(
502 current_iter,
503 l.end( ),
504 [ &new_value, &compare_pred ]( auto v ) {
505 return compare_pred( *v, new_value );
506 } );
507
508 if ( i != l.end( ) )
509 {
510 // foword hit
511 index_t const work_index = std::distance( l.begin( ), i );
512 if ( work_index != new_index )
513 move_item( work_index, new_index );
514 updater( *l[ new_index ], new_value );
515 }
516 else
517 {
518 // new value insert
519 insert_item( new_index, converter( std::move( new_value ) ) );
520 }
521 current_iter = l.begin( ) + new_index + 1u;
522 } // for new_index
523
524
525 // resize phase
526 if ( l.size( ) > new_list.size( ) )
527 {
528 auto const remove_count = l.size( ) - new_list.size( );
529 for ( auto i = 0u; i < remove_count; ++i )

531 remove_item( l.size( ) - 1u );
