22template< typename F, typename T >
23concept unmanaged_resource_update_handle = requires(
24 F h, std::function< void( std::shared_ptr< T > ) > handle ) { handle = h; };

27template< typename T >
28concept glab_notify_unmanaged_resource =
29 requires( T t, std::function< void( std::shared_ptr< T > ) > handle ) {
30 t.resource_glab_notify( ).connect( handle );
31 };
32template< typename T, typename Subject >
33concept instance_proxiable =
34 requires( T t, std::shared_ptr< Subject > ptr ) {
35 { t.swap_instance( ptr ) };
36 { t.instance( ) } -> std::same_as< std::shared_ptr< Subject >& >;
37 };
38template< typename T >
39concept glab_notify_unmanaged_resource_proxy =
40 requires( T t ) {
41 requires glab_notify_unmanaged_resource< T >;
42 typename T::subject_t;
43 requires instance_proxiable< T, typename T::subject_t >;
44 };
