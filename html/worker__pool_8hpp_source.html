173 void proc_worker( )
174 {
175 wait_initialize( );
176
177 while ( true )
178 {
179 bool is_assigned = false;
180 {
181 // ローカルキュー末尾からの取り出しを優先する
182 std::unique_lock< std::mutex > lock( mutex_ );
183 if ( !local_queue_.empty( ) )
184 {
185 current_task_ = local_queue_.back( );
186 local_queue_.pop_back( );
187 is_assigned = true;
188 }
189 }
190 if ( !is_assigned )
191 {
192 auto task = parent_->steal_or_pull( index_ );
193 if ( !task )
194 {
195 std::unique_lock< std::mutex > lock( mutex_ );
196 current_task_ = { };
197 cond_
198 .notify_all( ); // 何も処理していないことを通知する
199 cond_.wait( lock, [ & ]( ) {
200 // この述語内ではロックを取得しているはず
201 if ( is_requested_termination ) { return true; }
202 // steal_or_pullの時にロックを保持しているとデッドロックを起こす
203 lock.unlock( );
204 auto task = parent_->steal_or_pull( index_ );
205 lock.lock( );
206 // current_task_の更新時に再度ロックを取得する
207 current_task_ = std::move( task );
208 return !!current_task_;
209 } );
210 if ( is_requested_termination ) { break; }
211 }
212 else
213 {
214 std::unique_lock< std::mutex > lock( mutex_ );
215 current_task_ = std::move( task );
216 }
217 }
218 current_task_( );
219 }
