663template < typename T >
664struct sub_class_access;
665template < typename... Args >
666 requires( obserable_aggregate_entry< Args > && ... )
667struct sub_class_access< std::tuple< Args... > >
668{
669 using head_entyr_map_t = std::remove_cvref_t<
670 decltype( std::get< 0 >( std::tuple< Args... >{ } ) ) >;
671 using head_t = head_entyr_map_t::class_t;
672
673 using destination_entry_map_t
674 = std::remove_cvref_t< decltype( std::get< sizeof...( Args ) - 1u >(
675 std::tuple< Args... >{ }
676 ) ) >;
677 using type = std::remove_cvref_t<
678 typename destination_entry_map_t::return_t >::element_type;

680 template < typename H >
681 static constexpr auto
682 chain( H )
683 {
684 using current_t = H::class_t;
685 using return_shared_t = H::return_t;
686 return []( std::shared_ptr< current_t > const& c ) -> return_shared_t
687 { return H::get( *c ); };
688 }
689 template < typename H, typename HN, typename... Ts >
690 static constexpr auto
691 chain( H&&, HN&&, Ts&&... )
692 {
693 using current_t = H::class_t;
694 // using return_shared_t = H::return_t;
695 return []( std::shared_ptr< current_t > const& c )
696 {
697 auto next = chain( HN{ }, Ts{ }... );
698 return next( H::get( *c ) );
699 };
700 }
701
702 constexpr auto
703 get( std::shared_ptr< head_t > const& h ) const
704 {
705 auto func = chain( Args{ }... );
706 return func( h );
707 }
