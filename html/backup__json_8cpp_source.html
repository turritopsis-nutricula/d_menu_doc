54 /**
55 * @brief バッファ自動調整json読み込み
56 *
57 * バッファを倍々に調節しつつ読み込む
58 * バッファサイズ上限はmax_read_buffer_sizeまで
59 *
60 * @tparam N バッファサイズ係数。4[kbyte] * 2^N
61 * @tparam dto_type 読み込むdto型
62 */
63 //----------------------------------------------------------------------//
64 template< std::size_t N, typename dto_type >
65 struct backup_dto_reader
66 {
67 static constexpr std::size_t max_read_buffer_size =
68 4096u * 64; // 258[kbyte] = 4[kbyte] * 64
69
70 /**
71 * @param json json文字列
72 * @param dto_buffer 読み込む先DTO
73 */
74 auto operator( )( std::string const& json, dto_type& dto_buffer )
75 requires( N < max_read_buffer_size )
76 {
77 constexpr std::size_t buffer_size = N;
78 constexpr std::size_t next_size = buffer_size * 2u;
79
80 if ( json.size( ) > buffer_size - 8u )
81 {
82 if constexpr ( std::is_invocable_v<
83 backup_dto_reader< next_size, dto_type >,
84 std::string const&,
85 formats::latest::backup_json& > )
86 return backup_dto_reader< next_size, dto_type >{ }(
87 json,
88 dto_buffer );
89 }
90
91 using glz_read_buffer_t =
92 glz::basic_istream_buffer< std::istream, buffer_size >;
93
94 auto ss = std::stringstream( json );
95 glz_read_buffer_t buffer( ss );
96 return glz::read< detail::required_opt >( dto_buffer, buffer );
97 }
98 };

364//! @brief json文字列からbackup DTOに変換
365auto
366read_backup_json_impl( std::string const json_buf )
367 -> std::expected< formats::latest::backup_json, error_ctx >
368{
369
370 auto const version = read_backup_version( std::stringstream( json_buf ) );
371 if ( !version ) return std::unexpected( version.error( ) );
372
373 // latest road
374 formats::latest::backup_json newst_buffer;
375 if ( *version == formats::get_latest_format_version( ) )
376 {
377 auto error =
378 detail::backup_dto_reader< 1024u, formats::latest::backup_data >{ }(
379 json_buf,
380 newst_buffer );
381
382 if ( error ) return std::unexpected{ detail::make_error_code( error ) };
383 return newst_buffer;
384 }
385
386 constexpr std::size_t read_buffer_size = 4096u * 32u;
387 // 128[kbyte] = 4[kbyte] * 32
388 using glz_read_buffer_t =
389 glz::basic_istream_buffer< std::istream, read_buffer_size >;
390 using detail::format_convert::cascade_converter;
391
392 auto ss = std::stringstream( json_buf );
393 glz_read_buffer_t buffer( ss );
394 switch ( *version )
395 {
396 case 1 : // formats:v001
397 {
398 return cascade_converter< formats::v001::backup_json >{ }( buffer );
399 }
400 case 2 : // formats:v002
401 {
402 return cascade_converter< formats::v002::backup_json >{ }( buffer );
403 }
404 default :
405 {
406 break;
407 }
408 }
409 return newst_buffer;
410}
