70struct menu_query
71{
72
73 explicit menu_query( std::shared_ptr< dc::ConfigArg > conf_ptr )
74 : conf_ref_ptr_( conf_ptr )
75 { }
76
77 ddm::menu_id query_menu_id( ddm::menu_item_id id ) const
78 {
79 auto conf = conf_ref_ptr_.lock( );
80 if ( !conf ) return ddm::menu_id::nil_id( );
81
82 auto const& ml = conf->Menus( );
83
84 auto iter = std::ranges::find_if( ml, [ id ]( auto const& m ) -> bool {
85 auto const& l = m->MenuItems( );
86 return l.end( ) != ( std::ranges::find( l, id, []( auto const& i ) {
87 return i->get_ItemId( );
88 } ) );
89 } );
90
91 return ddm::menu_id( ( *iter )->get_MenuId( ) );
92 }
93
94 std::weak_ptr< dc::ConfigArg > conf_ref_ptr_;
95};

99// trigger_query

101//----------------------------------------------------------------------//
102struct trigger_query
103{
104
105
106 explicit trigger_query( std::weak_ptr< dc::ConfigArg > conf_ref_ptr )
107 : conf_ref_ptr_( std::move( conf_ref_ptr ) )
108 { }
109
110 static dd::triggers::key_input_modifier extract_modify(
111 dc::ConfGlobalKeyInputTrigger const& input )
112 {
113
114
115 int modifier = ( input.get_WithControlModifier( ) ? 0b0010 : 0 ) +
116 ( input.get_WithShiftModifier( ) ? 0b0100 : 0 ) +
117 ( input.get_WithAltModifier( ) ? 0b0001 : 0 ) +
118 ( input.get_WithWinModifier( ) ? 0b1000 : 0 );
119
120 return static_cast< dd::triggers::key_input_modifier >( modifier );
121 }
122
123 du::global_trigger_key_data get_trigger_data(
124 dd::triggers::trigger_id const id ) const
125 {
126 du::global_trigger_key_data data;
127 data.id_ = std::nullopt;
128
129 if ( auto conf_ptr = conf_ref_ptr_.lock( ) )
130 {
131 auto& l = conf_ptr->Triggers( );
132
133 auto iter = std::ranges::find( l, id.uuid( ), []( auto const& t ) {
134 return t->get_TriggerId( );
135 } );
136
137 if ( iter != l.end( ) )
138 {
139 auto const& tri = *( *iter );
140 auto const& key_input = *( tri.KeyInput_ptr( ) );
141 data.id_ = id;
142 data.key_code_ = static_cast< dd::triggers::virtual_key_code >(
143 key_input.get_KeyCode( ) );
144 data.modifier_ = extract_modify( key_input );
145 data.show_menu_id_ = dd::menu::menu_id( tri.get_ShowMenuId( ) );
146 data.is_active_ = tri.get_IsActive( );
147 }
148 }
149
150 return data;
151 }
152
153 std::weak_ptr< dc::ConfigArg > conf_ref_ptr_;
154};

315{
316 std::shared_ptr< observer_impl > impl_;

320
321public:
322 observer_wrapper(
323 std::shared_ptr< observer_impl > i,

325 std::shared_ptr< ConfigArg > view )
326 : impl_( i ), opt_id_( id ), view_ptr_( view )
327 { }
328
329private:
330 auto cfg_id( ) const
331 {
332 if ( auto v = view_ptr_.lock( ) )
333 return domain::app::config_id( v->get_ConfigId( ) );
334 else return domain::app::config_id( domain::id_tags::nil_uuid );
335 }
336
337 auto opt_id( ) const { return opt_id_; }
338 template< typename T >
339 requires std::
340 invocable< observer_impl const, std::remove_cvref_t< T > const& >
341 void relay( T const& c ) const
342 {
343 applications::run_worker( [ c, impl = impl_ ]( ) { ( *impl )( c ); } );
344 }
345
346 std::shared_ptr< ConfMenu > find_menu_from_item_id(
347 boost::uuids::uuid const& menu_item_id ) const
348 {
349 if ( auto view = view_ptr_.lock( ); !view ) { return nullptr; }
350 else
351 {
352 auto iter = std::ranges::find_if(
353 view->Menus( ), [ menu_item_id ]( auto const& m ) {
354 return std::ranges::find(
355 m->MenuItems( ), menu_item_id, []( auto const& i ) {
356 return i->get_ItemId( );
357 } ) != m->MenuItems( ).end( );
358 } );
359 if ( iter == view->Menus( ).end( ) ) return nullptr;
360
361 return ( *iter );
362 }
363 }
364 std::shared_ptr< ConfMenu > find_menu_from_item_id( ConfMenuItem const& i ) const
365 {
366 return find_menu_from_item_id( i.get_ItemId( ) );
367 }
368
369public:
370 // ConfMenu
371 void operator( )(
372 ConfMenu const& v,
373 changed_value< L"Name", std::string > arg ) const;
374
375 // ConfMenuItem
376 void operator( )(
377 ConfMenuItem const& v,

442template< typename observer_impl >
443void
444observer_wrapper< observer_impl >::operator( )(
445 ConfMenuItem const& v,
446 changed_value< L"Name", std::wstring > arg ) const
447{
448
449 if ( auto view = view_ptr_.lock( ) )
450 {
451 auto iter = std::ranges::find_if(
452 view->Menus( ), [ item_id = v.get_ItemId( ) ]( auto const& m ) {
453 return std::ranges::find(
454 m->MenuItems( ), item_id, []( auto const& i ) {
455 return i->get_ItemId( );
456 } ) != m->MenuItems( ).end( );
457 } );
458 if ( iter == view->Menus( ).end( ) ) return;
459
460 auto menu_id = ( *iter )->get_MenuId( );
461 relay(
462 usecase::dto::menu_item_update {
463 .cfg_id = cfg_id( ),
464 .menu_id = domain::menu::menu_id( menu_id ),
465 .item_id = domain::menu::menu_item_id( v.get_ItemId( ) ),
466 .name = util::strconv::wstring_to_u8string( arg.value( ) ) } );
467 }
468}

511/**
512 * @brief ConfMenuItem "ActionKind"
513 */
514//----------------------------------------------------------------------//
515template< typename O >
516void
517observer_wrapper< O >::operator( )(
518 ConfMenuItem const& v,
519 changed_value< L"ActionKind", BasicItemActionKind > arg ) const
520{
521 using domain::commands::common_command_kind;
522 auto kind = domain::commands::common_command_kind::none;
523 switch ( arg.value( ) )
524 {
525 case BasicItemActionKind::Nop :
526 {
527 kind = common_command_kind::none;
528 break;
529 }
530 case BasicItemActionKind::ShellExecuteAct :
531 {
532 kind = common_command_kind::shellexecute;
533 break;
534 }
535 case BasicItemActionKind::SubMenuShow :
536 kind = common_command_kind::open_sub_menu;
537 break;
538 case BasicItemActionKind::CustumAction :
539 kind = common_command_kind::custom_command;
540 break;
541 default : break;
542 }
543
544 auto menu = find_menu_from_item_id( v );
545
546
547 relay(
548 usecase::dto::common_action_select {
549 .option_id = opt_id( ),
550 .cfg_id = cfg_id( ),
551 .menu_id = domain::menu::menu_id(
552 menu ? menu->get_MenuId( ) : boost::uuids::nil_uuid( ) ),
553 .item_id = domain::menu::menu_item_id( v.get_ItemId( ) ),
554 .kind = kind,
555 } );
556}
