58 ddm::menu_id query_menu_id( ddm::menu_item_id id ) const
59 {
60 auto conf = conf_ref_ptr_.lock( );
61 if ( !conf ) return ddm::menu_id::nil_id( );
62
63 auto const& ml = conf->Menus( );
64
65 auto iter = std::ranges::find_if( ml, [ id ]( auto const& m ) -> bool {
66 auto const& l = m->MenuItems( );
67 return l.end( ) != ( std::ranges::find( l, id, []( auto const& i ) {
68 return i->get_ItemId( );
69 } ) );
70 } );
71
72 return ddm::menu_id( ( *iter )->get_MenuId( ) );
73 }

75 std::weak_ptr< dc::ConfigArg > conf_ref_ptr_;
76};

83struct trigger_query

86
87 explicit trigger_query( std::weak_ptr< dc::ConfigArg > conf_ref_ptr )
88 : conf_ref_ptr_( std::move( conf_ref_ptr ) )
89 { }
90
91 static dd::triggers::key_input_modifier extract_modify(
92 dc::ConfGlobalKeyInputTrigger const& input )
93 {
94
95
96 int modifier = ( input.get_WithControlModifier( ) ? 0b0010 : 0 ) +
97 ( input.get_WithShiftModifier( ) ? 0b0100 : 0 ) +
98 ( input.get_WithAltModifier( ) ? 0b0001 : 0 ) +
99 ( input.get_WithWinModifier( ) ? 0b1000 : 0 );
100
101 return static_cast< dd::triggers::key_input_modifier >( modifier );
102 }
103
104 du::global_trigger_key_data get_trigger_data(
105 dd::triggers::trigger_id const id ) const
106 {
107 du::global_trigger_key_data data;
108 data.id_ = std::nullopt;
109
110 if ( auto conf_ptr = conf_ref_ptr_.lock( ) )
111 {
112 auto& l = conf_ptr->Triggers( );
113
114 auto iter = std::ranges::find( l, id.uuid( ), []( auto const& t ) {
115 return t->get_TriggerId( );
116 } );
117
118 if ( iter != l.end( ) )
119 {
120 auto const& tri = *( *iter );
121 auto const& key_input = *( tri.KeyInput_ptr( ) );
122 data.id_ = id;
123 data.key_code_ = static_cast< dd::triggers::virtual_key_code >(
124 key_input.get_KeyCode( ) );
125 data.modifier_ = extract_modify( key_input );
126 data.show_menu_id_ = dd::menu::menu_id( tri.get_ShowMenuId( ) );
127 data.is_active_ = tri.get_IsActive( );
128 }
129 }
130
131 return data;
132 }

135};
