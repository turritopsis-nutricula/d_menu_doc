24// spin lock
25template< typename T >
26struct spin_lock_policy
27{
28 static void lock( std::atomic< T >& v, T delta, T test_bit ) noexcept
29 {
30 T expected;
31 do
32 {
33 do { expected = v.load( ); }
34 while ( expected & test_bit );
35 }
36 while ( !v.compare_exchange_weak( expected, expected + delta ) );
37 }
38
39 static void unlock( std::atomic< T >& v, T delta ) noexcept
40 {
41 v.fetch_sub( delta );
42 }
43};
44
45// wait lock
46template< typename T >
47struct wait_lock_policy
48{
49 static void lock( std::atomic< T >& v, T delta, T test_bit ) noexcept
50 {
51 T expected;
52 do
53 {
54 expected = v.load( );
55 while ( expected & test_bit )
56 {
57 v.wait( expected );
58 expected = v.load( );
59 }
60 }
61 while ( !v.compare_exchange_weak( expected, expected + delta ) );
62 }
63 static void unlock( std::atomic< T >& v, T delta ) noexcept
64 {
65 v.fetch_sub( delta );
66 v.notify_all( );
67 }
68};

71// upgradable mutex
72// ReadLockUpgradableでロックを取得した場合、
73// 他のスレッドからのReadLockを妨げずに共存しますが、
74// 他のスレッドからReadLockUpgradableを同時に取得することはできません。
75// １つのスレッドだけがReadLockUpgradableを取れるので、
76// Upgradeの段階で競合が発生しないことと、
77// Upgrade中に他のスレッドがライトロックを取得したりしないことが保障されます。
78//
79// ステート移動 立てるビット 待機するビット
80// アイドル→リード リード ライト
81// アイドル→ライト ライト 全ビット
82// アイドル→昇格可能リード 昇格可能リード ライト、昇格可能リード
83
84template< typename P >
85struct upgradable_mutex
86{
87 P* m_;
88 bool need_unlock = false;
89
90 upgradable_mutex( P& m ) noexcept : m_( &m ) { }
91 ~upgradable_mutex( ) noexcept { unlock( ); }
92 void lock( ) noexcept
93 {
94 m_->upgradable_lock( );
95 need_unlock = true;
96 }
97 void unlock( ) noexcept
98 {
99 if ( need_unlock )
100 {
101 m_->upgradable_unlock( );
102 need_unlock = false;
103 }
104 }
105};
106
107template< typename P >
108struct write_lock_mutex
109{
110 using up_mtx_t = upgradable_mutex< P >;
111 up_mtx_t* ul_;
112 bool need_unlock = false;
113
114 write_lock_mutex( up_mtx_t& ul ) : ul_( &ul ) { }
115 ~write_lock_mutex( ) { unlock( ); }
116
117 void lock( ) noexcept
118 {
119 assert( ul_->m_->is_upgradable_lock( ) );
120 ul_->m_->upgrade( );
121 ul_->need_unlock = false;
122 need_unlock = true;
123 }
124 void unlock( ) noexcept
125 {
126 if ( need_unlock )
127 {
128 ul_->m_->write_unlock( );
129 need_unlock = false;
130 }
131 }
132};

136template< typename LockPolicy >
137class upgrade_mutex_base
138{
139 std::atomic< int > v_ = 0;
140
141 static constexpr int write_lock_bit = 0x40000000;
142 static constexpr int upgradable_bit = 0x00010000;
143 static constexpr int reader_bits = 0x0000ffff;
144
145public:
146public:
147 void lock( ) noexcept { read_lock( ); }
148 void unlock( ) noexcept { read_unlock( ); }
149 void read_lock( ) noexcept { LockPolicy::lock( v_, 1, write_lock_bit ); }
150 void read_unlock( ) noexcept { LockPolicy::unlock( v_, 1 ); }
151 void upgradable_lock( ) noexcept
152 {
153 LockPolicy::lock( v_, upgradable_bit, write_lock_bit | upgradable_bit );
154 }
155 void upgradable_unlock( ) noexcept
156 {
157 LockPolicy::unlock( v_, upgradable_bit );
158 }
159 void upgrade( ) noexcept
160 {
161 LockPolicy::lock(
162 v_, write_lock_bit - upgradable_bit, reader_bits | write_lock_bit );
163 }
164 void write_lock( ) noexcept
165 {
166 LockPolicy::lock( v_, write_lock_bit, reader_bits | write_lock_bit );
167 }
168 void write_unlock( ) noexcept { LockPolicy::unlock( v_, write_lock_bit ); }
169
170 bool is_upgradable_lock( ) const noexcept
171 {
172 int value = v_.load( );
173 return value & upgradable_bit;
174 }
175 bool is_write_lock( ) const noexcept
176 {
177 int value = v_.load( );
178 return value & write_lock_bit;
179 }
180};

182//----------------------------------------------------------------------//
183// try upgradable mutex
184
185// ※
186// 万が一昇格に失敗した場合、次に出来ることはリードロックを解除することだけです。
187// もし、どうしてもライトロックが欲しければ一度リードロックも解除して
188// 他のスレッドにライトロックを譲った後、改めてライトロックの取得を試みるしかありません。
189
190// TryUpgrade関数のwhileループで昇格中ステートへの移行を試みます。
191// 他のスレッドに先に昇格中ビットを立てられた場合は失敗します。
192//
193// 自身によって昇格中ビットを立てることに成功すればLockInternalで
194// ライトステートへ移行します。readerBitsの最下位ビットを待機対象から
195// 除外するのは、自身がリードロックを持っているため
196// リードロックのカウンタが1の時は通過させなければいけないためです。
197//
198// ステート移動 立てるビット 待機するビット 備考
199// アイドル→リード リード ライト、昇格中
200// アイドル→ライト ライト 全ビット
201// リード→昇格中 昇格中 無し 他のスレッドが昇格中ビットを取得したら失敗
202// 昇格中→ライト ライト ライト、リード
203// リードロックの１つは自分が持っているので、最下位ビットを待機対象から除外
204
205template< typename P >
206struct read_lock_mutex
207{
208 P* m_;
209 bool is_need_unlock_ = false;
210
211 explicit read_lock_mutex( P& m ) : m_( &m ) { }
212 ~read_lock_mutex( ) { unlock( ); }
213
214 void lock( )
215 {
216 m_->read_lock( );
217 is_need_unlock_ = true;
218 }
219 void unlock( )
220 {
221 if ( is_need_unlock_ )
222 {
223 m_->read_unlock( );
224 is_need_unlock_ = false;
225 }
226 }
227};

230template< typename LockPolicy >
231struct try_upgradable_mutex_base
232{
233 std::atomic< int > v_ = 0;
234 static const int writeLockBit = 0x40000000;
235 static const int upgradingBit = 0x20000000;
236 static const int readerBits = 0x0000ffff;
237
238public:
239 void lock( ) noexcept { read_lock( ); }
240
241 void unlock( ) noexcept { read_unlock( ); }
242
243 void read_lock( ) noexcept
244 {
245 LockPolicy::lock( v_, 1, writeLockBit | upgradingBit );
246 }
247
248 void read_unlock( ) noexcept { LockPolicy::unlock( v_, 1 ); }
249 void write_lock( ) noexcept
250 {
251 LockPolicy::lock( v_, writeLockBit, 0xffffffff );
252 }
253 void write_unlock( ) noexcept { LockPolicy::unlock( v_, writeLockBit ); }
254
255 bool try_upgrade( ) noexcept
256 {
257 int expected;
258 do
259 {
260 expected = v_;
261 if ( expected & upgradingBit ) { return false; }
262 }
263 while (
264 !v_.compare_exchange_weak( expected, expected | upgradingBit ) );
265 LockPolicy::lock(
266 v_,
267 writeLockBit - upgradingBit - 1,
268 writeLockBit | ( readerBits & ~1 ) );
269
270 return true;
271 }
272};
