25template < typename T >
26struct spin_lock_policy
27{
28 static void
29 lock( std::atomic< T >& v, T delta, T test_bit ) noexcept
30 {
31 T expected;
32 do
33 {
34 do
35 {
36 expected = v.load( );
37 } while ( expected & test_bit );
38 } while ( !v.compare_exchange_weak( expected, expected + delta ) );
39 }
40
41 static void
42 unlock( std::atomic< T >& v, T delta ) noexcept
43 {
44 v.fetch_sub( delta );
45 }
46};
47
48// wait lock
49template < typename T >
50struct wait_lock_policy
51{
52 static void
53 lock( std::atomic< T >& v, T delta, T test_bit ) noexcept
54 {
55 T expected;
56 do
57 {
58 expected = v.load( );
59 while ( expected & test_bit )
60 {
61 v.wait( expected );
62 expected = v.load( );
63 }
64 } while ( !v.compare_exchange_weak( expected, expected + delta ) );
65 }
66 static void
67 unlock( std::atomic< T >& v, T delta ) noexcept
68 {
69 v.fetch_sub( delta );
70 v.notify_all( );
71 }
72};

75// upgradable mutex
76// ReadLockUpgradableでロックを取得した場合、
77// 他のスレッドからのReadLockを妨げずに共存しますが、
78// 他のスレッドからReadLockUpgradableを同時に取得することはできません。
79// １つのスレッドだけがReadLockUpgradableを取れるので、
80// Upgradeの段階で競合が発生しないことと、
81// Upgrade中に他のスレッドがライトロックを取得したりしないことが保障されます。
82//
83// ステート移動 立てるビット 待機するビット
84// アイドル→リード リード ライト
85// アイドル→ライト ライト 全ビット
86// アイドル→昇格可能リード 昇格可能リード ライト、昇格可能リード
87
88template < typename P >
89struct upgradable_mutex
90{
91 P* m_;
92 bool need_unlock = false;
93
94 upgradable_mutex( P& m ) noexcept : m_( &m ) { }
95 ~upgradable_mutex( ) noexcept { unlock( ); }
96 void
97 lock( ) noexcept
98 {
99 m_->upgradable_lock( );
100 need_unlock = true;
101 }
102 void
103 unlock( ) noexcept
104 {
105 if ( need_unlock )
106 {
107 m_->upgradable_unlock( );
108 need_unlock = false;
109 }
110 }
111};
112
113template < typename P >
114struct write_lock_mutex
115{
116 using up_mtx_t = upgradable_mutex< P >;
117 up_mtx_t* ul_;
118 bool need_unlock = false;
119
120 write_lock_mutex( up_mtx_t& ul ) : ul_( &ul ) { }
121 ~write_lock_mutex( ) { unlock( ); }
122
123 void
124 lock( ) noexcept
125 {
126 assert( ul_->m_->is_upgradable_lock( ) );
127 ul_->m_->upgrade( );
128 ul_->need_unlock = false;
129 need_unlock = true;
130 }
131 void
132 unlock( ) noexcept
133 {
134 if ( need_unlock )
135 {
136 ul_->m_->write_unlock( );
137 need_unlock = false;
138 }
139 }
140};

144template < typename LockPolicy >
145class upgrade_mutex_base
146{
147 std::atomic< int > v_ = 0;
148
149 static constexpr int write_lock_bit = 0x40000000;
150 static constexpr int upgradable_bit = 0x00010000;
151 static constexpr int reader_bits = 0x0000ffff;
152
153 public:
154 public:
155 void
156 lock( ) noexcept
157 {
158 read_lock( );
159 }
160 void
161 unlock( ) noexcept
162 {
163 read_unlock( );
164 }
165 void
166 read_lock( ) noexcept
167 {
168 LockPolicy::lock( v_, 1, write_lock_bit );
169 }
170 void
171 read_unlock( ) noexcept
172 {
173 LockPolicy::unlock( v_, 1 );
174 }
175 void
176 upgradable_lock( ) noexcept
177 {
178 LockPolicy::lock( v_, upgradable_bit, write_lock_bit | upgradable_bit );
179 }
180 void
181 upgradable_unlock( ) noexcept
182 {
183 LockPolicy::unlock( v_, upgradable_bit );
184 }
185 void
186 upgrade( ) noexcept
187 {
188 LockPolicy::lock(
189 v_, write_lock_bit - upgradable_bit, reader_bits | write_lock_bit
190 );
191 }
192 void
193 write_lock( ) noexcept
194 {
195 LockPolicy::lock( v_, write_lock_bit, reader_bits | write_lock_bit );
196 }
197 void
198 write_unlock( ) noexcept
199 {
200 LockPolicy::unlock( v_, write_lock_bit );
201 }
202
203 bool
204 is_upgradable_lock( ) const noexcept
205 {
206 int value = v_.load( );
207 return value & upgradable_bit;
208 }
209 bool
210 is_write_lock( ) const noexcept
211 {
212 int value = v_.load( );
213 return value & write_lock_bit;
214 }
215};

218// try upgradable mutex
219
220// ※
221// 万が一昇格に失敗した場合、次に出来ることはリードロックを解除することだけです。
222// もし、どうしてもライトロックが欲しければ一度リードロックも解除して
223// 他のスレッドにライトロックを譲った後、改めてライトロックの取得を試みるしかありません。
224
225// TryUpgrade関数のwhileループで昇格中ステートへの移行を試みます。
226// 他のスレッドに先に昇格中ビットを立てられた場合は失敗します。
227//
228// 自身によって昇格中ビットを立てることに成功すればLockInternalで
229// ライトステートへ移行します。readerBitsの最下位ビットを待機対象から
230// 除外するのは、自身がリードロックを持っているため
231// リードロックのカウンタが1の時は通過させなければいけないためです。
232//
233// ステート移動 立てるビット 待機するビット 備考
234// アイドル→リード リード ライト、昇格中
235// アイドル→ライト ライト 全ビット
236// リード→昇格中 昇格中 無し 他のスレッドが昇格中ビットを取得したら失敗
237// 昇格中→ライト ライト ライト、リード
238// リードロックの１つは自分が持っているので、最下位ビットを待機対象から除外
239
240template < typename P >
241struct read_lock_mutex
242{
243 P* m_;
244 bool is_need_unlock_ = false;
245
246 explicit read_lock_mutex( P& m ) : m_( &m ) { }
247 ~read_lock_mutex( ) { unlock( ); }
248
249 void
250 lock( )
251 {
252 m_->read_lock( );
253 is_need_unlock_ = true;
254 }
255 void
256 unlock( )
257 {
258 if ( is_need_unlock_ )
259 {
260 m_->read_unlock( );
261 is_need_unlock_ = false;
262 }
263 }
264};

267template < typename LockPolicy >
268struct try_upgradable_mutex_base
269{
270 std::atomic< int > v_ = 0;
271 static const int writeLockBit = 0x40000000;
272 static const int upgradingBit = 0x20000000;
273 static const int readerBits = 0x0000ffff;
274
275 public:
276 void
277 lock( ) noexcept
278 {
279 read_lock( );
280 }
281
282 void
283 unlock( ) noexcept
284 {
285 read_unlock( );
286 }
287
288 void
289 read_lock( ) noexcept
290 {
291 LockPolicy::lock( v_, 1, writeLockBit | upgradingBit );
292 }
293
294 void
295 read_unlock( ) noexcept
296 {
297 LockPolicy::unlock( v_, 1 );
298 }
299 void
300 write_lock( ) noexcept
301 {
302 LockPolicy::lock( v_, writeLockBit, 0xffffffff );
303 }
304 void
305 write_unlock( ) noexcept
306 {
307 LockPolicy::unlock( v_, writeLockBit );
308 }
309
310 bool
311 try_upgrade( ) noexcept
312 {
313 int expected;
314 do
315 {
316 expected = v_;
317 if ( expected & upgradingBit )
318 {
319 return false;
320 }
321 } while ( !v_.compare_exchange_weak( expected, expected | upgradingBit )
322 );
323 LockPolicy::lock(
324 v_,
325 writeLockBit - upgradingBit - 1,
326 writeLockBit | ( readerBits & ~1 )
327 );
328
329 return true;
330 }
331};

381try_write_lock( detail_util::read_lock_mutex< P >& m, F&& f ) noexcept
382{
383 assert( m.is_need_unlock_ );
384 if ( m.m_->try_upgrade( ) )
385 {
386 m.is_need_unlock_ = false;
387 f( );
388 m.m_->write_unlock( );
389 return true;
390 }
