107 std::wstring generate_command_name(
108 command_version_t ver,
109 command_index_t index,
110 domain::commands::parameter_pack const& pack )
111 {
112 std::unique_ptr< command_name_generator_interface > gen;
113 {
114 command_name_generator_interface* ptr = nullptr;
115 this->get_name_generator( ver, index, &ptr );
116 if ( ptr != nullptr )
117 gen.reset( ptr );
118 else
119 return L"* not set command name *";
120 }
121
122 auto param_index = 0u;
123 for ( auto&& p : pack )
124 {
125 std::visit( [ &gen, &param_index ]( auto const& v ) mutable {
126 paramter_kind_t k = v.kind;
127 auto value = parameter_converter{ }( v.value );
128
129 if constexpr ( !std::is_same_v<
130 std::remove_cvref_t< decltype( value ) >,
131 std::monostate > )
132 gen->set_parameter( param_index++, k, value );
133 }, p );
134 }
135
136 return { gen->generate( ) };
137 }
